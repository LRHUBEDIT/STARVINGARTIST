function Library:Refresh(The_Drop, newtable)
    if The_Drop.Clear and The_Drop.Add then
        The_Drop:Clear()
        if newtable then
            for i, v in pairs(newtable) do
                The_Drop:Add(tostring(v))
            end
        end
    end
end
function Get_Type(ID)
    local IsPass =
        pcall(
        function()
            game:GetService("MarketplaceService"):GetProductInfo(ID, "GamePass")
        end
    )

    if IsPass then
        return tostring("Pass")
    else
        for i, v in pairs(game:GetService("MarketplaceService"):GetProductInfo(ID)) do
            return tostring("Shirt")
        end
    end
end

function Hop()
    local Job
    for i, v in pairs(
        game:GetService("HttpService"):JSONDecode(
            game:HttpGet(
                "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"
            )
        ).data
    ) do
        if type(v) == "table" and v.maxPlayers > v.playing and v.id ~= game.JobId then
            Job = v.id
        end
    end
    if not Job then
        return Hop()
    end
    game:GetService("TeleportService"):TeleportToPlaceInstance(game.PlaceId, Job)
end

function Get_Enemy(Thing)
    if Current_Game == "Plinko" then task.wait(1) end
    for i,v in pairs(workspace.ArenasREAL:GetChildren()) do
        if v:FindFirstChild("ArenaTemplate") and v.ArenaTemplate:FindFirstChild("Important") and v.ArenaTemplate:FindFirstChild("Red") and v.ArenaTemplate:FindFirstChild("Blue") then
            if Thing == "Color" then
                if tostring(v.ArenaTemplate.Blue.Character:WaitForChild("Nametag").Frame.Username.Text) == "@"..tostring(game:GetService("Players").LocalPlayer.Name) then
                    return "Blue"
                elseif tostring(v.ArenaTemplate.Red.Character:WaitForChild("Nametag").Frame.Username.Text) == "@"..tostring(game:GetService("Players").LocalPlayer.Name) then
                    return "Red"
                end
            elseif Thing == "Username" and tostring(v.ArenaTemplate.Blue.Character:WaitForChild("Nametag").Frame.Username.Text) == "@"..tostring(game:GetService("Players").LocalPlayer.Name) or tostring(v.ArenaTemplate.Red.Character:WaitForChild("Nametag").Frame.Username.Text) == "@"..tostring(game:GetService("Players").LocalPlayer.Name) then
                if tostring(v.ArenaTemplate.Blue.Character:WaitForChild("Nametag").Frame.Username.Text) ~= "@"..tostring(game:GetService("Players").LocalPlayer.Name) then
                    return v.ArenaTemplate.Blue.Character:WaitForChild("Nametag").Frame.Username.Text:gsub("@","")
                elseif tostring(v.ArenaTemplate.Red.Character:WaitForChild("Nametag").Frame.Username.Text) ~= "@"..tostring(game:GetService("Players").LocalPlayer.Name) then
                    return v.ArenaTemplate.Red.Character:WaitForChild("Nametag").Frame.Username.Text:gsub("@","")
                end
            end
        end
    end
end

function Get_Closest_Enemy()
    for i,v in pairs(game:GetService("Players"):GetChildren()) do
        if v.Character and v.Character:FindFirstChild("HumanoidRootPart") and v.Character:FindFirstChild("Humanoid") and v.Character.Humanoid.Health > 0 and v.Name ~= game:GetService("Players").LocalPlayer.Name then
            if v.Name == tostring(Get_Enemy("Username")) then
                return v
            end
        end
    end
end

function Get_Cup_Folder(Arena)
    for i,v in pairs(Arena.Important:GetChildren()) do
        if v.Name ~= "CupPositions" and v.Name ~= "Diamonds" then
            return v
        end
    end
end
